        // 3. Функция генерации иконок-табов
        function renderTabs() {
            const tabsContainer = document.getElementById('tabs-container');
            tabsContainer.innerHTML = ''; // Очищаем

            laptopsData.forEach(laptop => {
                const tab = document.createElement('div');
                tab.className = `tab-icon ${laptop.id === activeLaptopId ? 'active' : ''}`;
                tab.onclick = () => selectLaptop(laptop.id);

                // Вставляем схематичную SVG иконку ноутбука, чтобы не зависеть от внешних картинок
                tab.innerHTML = `
                    <svg viewBox="0 0 24 24">
                        <path d="M20,18H4V6H20M20,4H4C2.89,4 2,4.89 2,6V18A2,2 0 0,0 4,20H20A2,2 0 0,0 22,18V6C22,4.89 21.1,4 20,4M24,22H0V24H24V22Z" />
                    </svg>
                    <span style="font-weight:600; font-size:14px;">${laptop.iconText}</span>
                `;
                tabsContainer.appendChild(tab);
            });
        }

        // 4. Функция обновления информации в карточке товара
        function updateProductCard() {
            const cardElement = document.getElementById('product-card');
            // Убираем класс анимации для эффекта "перезагрузки" контента
            cardElement.classList.remove('show');

            // Находим данные выбранного ноутбука
            const laptop = laptopsData.find(l => l.id === activeLaptopId);
            const content = laptop[currentLanguage];

            // Заполняем данные на выбранном языке
            document.getElementById('product-title').innerText = content.title;
            document.getElementById('product-desc').innerText = content.desc;
            document.getElementById('product-price').innerText = content.price;

            // Рендерим характеристики списком
            const specsContainer = document.getElementById('product-specs');
            specsContainer.innerHTML = `
                <li>${content.cpu}</li>
                <li>${content.ram}</li>
                <li>${content.storage}</li>
            `;

            // Меняем внешний вид нашей "картинки-заглушки" (градиент и текст)
            const imgMock = document.getElementById('product-img');
            imgMock.style.background = laptop.color;
            imgMock.innerText = laptop.iconText + " Display";

            // Запускаем плавное появление с задержкой, чтобы анимация сработала красиво
            setTimeout(() => {
                cardElement.classList.add('show');
            }, 50);
        }

        // 5. Логика переключения ноутбуков при клике на иконки
        function selectLaptop(id) {
            activeLaptopId = id;
            renderTabs();       // Перерисовываем табы (чтобы сменить активную иконку)
            updateProductCard(); // Обновляем информацию
        }

        // 6. Логика смены языка сайта (RU / EN)
        function toggleLanguage() {
            currentLanguage = currentLanguage === "ru" ? "en" : "ru";
            
            // Меняем текст на кнопке переключения
            document.getElementById('lang-btn').innerText = currentLanguage === "ru" ? "EN" : "RU";
            
            // Обновляем карточку с новым языком
            updateProductCard();
        }
